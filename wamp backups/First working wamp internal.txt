# First functional VHost
<VirtualHost 192.168.1.7:80>
	ServerAdmin ydavidf@gmail.com
    DocumentRoot "c:/wamp/www/xblinternal"
	ServerName xbl.local
    ServerAlias www.xbl.local
    ErrorLog "logs/localhost-error.log"
    CustomLog "logs/localhost-access.log" common
	<Directory "C:/wamp/www/xblinternal">
		AllowOverride All
		Require local
	</Directory>
</VirtualHost>
